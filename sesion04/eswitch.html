Uso de la estructura de control Switch

      
      let suma = 1000 + 1000;
      switch (suma) {
        case <= 100:
          document.getElementById('resultado').innerHTML = 'Necesita aumentar monto';
          break;
        case <= 300:
          document.getElementById('resultado').innerHTML = 'Valor minimo esperado';
          break;
        case <= 500:
          document.getElementById('resultado').innerHTML = 'Informe a Gerencia';
          break;
        default:
          document.getElementById('resultado').innerHTML = 'Resultado fuera de rango';
      }

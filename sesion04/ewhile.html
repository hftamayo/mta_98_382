Uso de la estructura de control While

      
        let limite = 15;
        let suma = 0;
        let contador = 1;
        let arreglo = [15];

        while(contador <= limite) {
          suma += contador;
          arreglo[contador-1] = contador;
          contador++;
        }
        document.getElementById('resultado').innerHTML = arreglo + " = " +suma;

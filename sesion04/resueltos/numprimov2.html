Calcular el factorial de un nuemro

      
        const numero = 5;
        let divisor = 0;
        let esPrimo = true;
        let arrOperaciones = new Array();
        for(divisor = 1; divisor <= numero; divisor++){
          if(numero % divisor ==0){
                esPrimo = false;
                arrOperaciones[divisor - 1] = (numero + " % " + divisor + " = " + (numero % divisor)) + "-";
            }
            else{
                arrOperaciones[divisor -1 ] = (numero + " % " + divisor + " = " + (numero % divisor)) + "-";
            }
        }
        document.getElementById('resultado').innerHTML = esPrimo + " iteraciones: " + arrOperaciones;

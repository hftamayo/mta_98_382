Determinacion del ranking academico

      
        const notaFinal = 7.5;
        let ranking = "";
        if( notaFinal >= 9 ){
          ranking = "Muy Bueno";
        }
        else if( notaFinal >= 7 ){
          ranking = "bueno";
        }
        else if( notaFinal >= 6 ){
          ranking = "necesita mejorar";
        }
        else if( notaFinal >= 5){
          ranking = "por debajo de lo esperado"
        }
        else{
          ranking = "reprobado";
        }

        document.getElementById('resultado').innerHTML = ranking;

Calcular el factorial de un nuemro

      
        const numero = 23;
        let divisor = 1;
        let esPrimo = true;
        let arrOperaciones = new Array();
        let i = 0;
        do{
            if(numero % divisor ==0){
                esPrimo = false;
                arrOperaciones[i] = (numero + " % " + divisor + " = " + (numero % divisor)) + "-";
            }
            else{
                arrOperaciones[i] = (numero + " % " + divisor + " = " + (numero % divisor)) + "-";
            }
            divisor++;
            i++;
        }while(divisor <= numero)
    
        //por que esPrimo siempre cambia a false?
    
        document.getElementById('resultado').innerHTML = esPrimo + " iteraciones: " + arrOperaciones;

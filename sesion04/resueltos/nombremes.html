Uso de la estructura de control Switch

      
      const mesNumero = 10;
      switch (mesNumero) {
        case 1:
          document.getElementById('resultado').innerHTML = 'Enero';
          break;
          case 2:
          document.getElementById('resultado').innerHTML = 'Febrero';
          break;
          case 3:
          document.getElementById('resultado').innerHTML = 'Marzo';
          break;
          case 4:
          document.getElementById('resultado').innerHTML = 'Abril';
          break;
          case 5:
          document.getElementById('resultado').innerHTML = 'Mayo';
          break;
          case 6:
          document.getElementById('resultado').innerHTML = 'Junio';
          break;
          case 7:
          document.getElementById('resultado').innerHTML = 'Julio';
          break;
          case 8:
          document.getElementById('resultado').innerHTML = 'Agosto';
          break;
          case 9:
          document.getElementById('resultado').innerHTML = 'Septiembre';
          break;
          case 10:
          document.getElementById('resultado').innerHTML = 'Octubre';
          break;
          case 11:
          document.getElementById('resultado').innerHTML = 'Noviembre';
          break;
          case 12:
          document.getElementById('resultado').innerHTML = 'Diciembre';
          break;
        default:
          document.getElementById('resultado').innerHTML = 'Numero fuera de rango';
      }

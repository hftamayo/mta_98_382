Calcular el factorial de un numero v2

      
        const numero = 5;
        let factorial = 1;
        let arrFactores = new Array();
        for(let i = 1; i <= numero; i++){
          arrFactores[i-1] = (factorial + "*" + i + "=" + (factorial * i));
          factorial *= i;
        }
        document.getElementById('resultado').innerHTML = factorial + " iteraciones: " + arrFactores;
